28std::optional<std::any> print(const std::vector<std::any> &args,
29 Environment *fnEnv) {
30 std::any toPrint{args[0]};
31 if(toPrint.type() == typeid(std::string))
32 std::cout << std::any_cast<std::string>(toPrint) << '\n';
33 else if(toPrint.type() == typeid(bool))
34 std::cout << (std::any_cast<bool>(toPrint) ? "true" : "false") << '\n';
35 else if(toPrint.type() == typeid(long double))
36 std::cout << std::any_cast<long double>(toPrint) << '\n';
37 return {};
38}
