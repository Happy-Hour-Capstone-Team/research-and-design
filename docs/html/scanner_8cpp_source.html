15void Scanner::scanToken() {
16 switch(text[pos]) {
17 case ' ':
18 case '\r':
19 case '\t': break;
20 case '{': addToken("{", Token::Type::LeftCurly); break;
21 case '}': addToken("}", Token::Type::RightCurly); break;
22 case ';': addToken(";", Token::Type::Semicolon); break;
23 case '(': addToken("(", Token::Type::LeftParen); break;
24 case ')': addToken(")", Token::Type::RightParen); break;
25 case '*': addToken("*", Token::Type::Asterisk); break;
26 case '+': addToken("+", Token::Type::Plus); break;
27 case '-': addToken("-", Token::Type::Dash); break;
28 case ',': addToken(",", Token::Type::Comma); break;
29 case '.': addToken(".", Token::Type::Dot); break;
30 case ':': addToken(":", Token::Type::Colon); break;
31 case '!':
32 if(text[pos + 1] == '=')
33 addToken("!=", Token::Type::NotEqualTo);
34 else
35 addToken("!", Token::Type::Exclamation);
36 break;
37 case '=':
38 if(text[pos + 1] == '=')
39 addToken("==", Token::Type::EqualTo);
40 else
41 addToken("=", Token::Type::Equal);
42 break;
43 case '<':
44 if(text[pos + 1] == '=')
45 addToken("<=", Token::Type::LessThanOrEqualTo);
46 else
47 addToken("<", Token::Type::LessThan);
48 break;
49 case '>':
50 if(text[pos + 1] == '=')
51 addToken(">=", Token::Type::GreaterThanOrEqualTo);
52 else
53 addToken(">", Token::Type::GreaterThan);
54 break;
55 case '\n': newLine(); break;
56 case '/': forwardSlash(); break;
57 case '"': string(); break;
58 default: longTokens(); break;
59 };

62void Scanner::forwardSlash() {
63 switch(text[pos + 1]) {
64 case '/':
65 while(text[pos + 1] != '\n') pos++;
66 break;
67 case ':':
68 while(pos + 2 < text.length() &&
69 (text[pos + 1] != ':' || text[pos + 2] != '/')) {
70 if(text[pos] == '\n') newLine();
71 incPosCol();
72 }
73 incPosCol(2);
74 break;
75 default: addToken("/", Token::Type::ForwardSlash); break;
76 };
