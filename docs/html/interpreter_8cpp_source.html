98std::optional<std::any> Interpreter::visit(const Expression::Binary &binary,
99 Environment *env) {
100 std::any leftVal = evaluate(binary.left.get(), env);
101 std::any rightVal = evaluate(binary.right.get(), env);
102 try {
103 if(leftVal.type() == typeid(std::string))
104 return stringOperation(std::any_cast<std::string>(leftVal),
105 binary.op.type,
106 std::any_cast<std::string>(rightVal));
107 else if(leftVal.type() == typeid(bool))
108 return booleanOperation(std::any_cast<bool>(leftVal),
109 binary.op.type,
110 std::any_cast<bool>(rightVal));
111 else
112 return numericOperation(std::any_cast<long double>(leftVal),
113 binary.op.type,
114 std::any_cast<long double>(rightVal));
115 } catch(std::bad_any_cast) {
116 throw std::runtime_error("Type mismatch between operator!");
117 }

145std::optional<std::any> Interpreter::visit(const Expression::Call &call,
146 Environment *env) {
147 std::any callee{evaluate(call.callee.get(), env)};
148 std::vector<std::any> args{};
149 for(std::size_t i{0}; i < call.args.size(); i++)
150 args.push_back(evaluate(call.args[i].get(), env));
151 try {
152 Callable callable{std::any_cast<Callable>(callee)};
153 if(args.size() < callable.minArity || args.size() > callable.maxArity)
154 throw std::runtime_error{
155 "Method expected at least " + std::to_string(callable.minArity) +
156 " arguments, at most " + std::to_string(callable.maxArity) +
157 " arguments, and received " + std::to_string(args.size()) +
158 " arguments."};
159 return callable.procedure(args, callable.fnEnv.get());
160 } catch(std::bad_any_cast) {
161 try {
162 Prototypable prototype{std::any_cast<Prototypable>(callee)};
163 Prototypable newPrototype{prototype.copy()};
164 if(args.size() < newPrototype.constructor.minArity ||
165 args.size() > newPrototype.constructor.maxArity)
166 throw std::runtime_error{
167 "Constructor expected at least " +
168 std::to_string(newPrototype.constructor.minArity) +
169 " arguments, at most " +
170 std::to_string(newPrototype.constructor.maxArity) +
171 " arguments, and received " + std::to_string(args.size()) +
172 " arguments."};
173 newPrototype.constructor.procedure(args, newPrototype.methodEnv.get());
174 return newPrototype;
175 } catch(std::bad_any_cast) {
176 throw std::runtime_error{"Only functions and prototypes may be called."};
177 }
178 }

181std::optional<std::any> Interpreter::visit(const Expression::Lambda &lambda,
182 Environment *env) {
183 Procedure lambdaFn = [&lambda, this](const std::vector<std::any> &args,
184 Environment *fnEnv) {
185 std::unique_ptr<Environment> scopedEnv{
186 std::make_unique<Environment>(fnEnv)};
187 for(std::size_t i{0};
188 i < lambda.params.size() + lambda.defaultParams.size();
189 i++) {
190 if(i < lambda.params.size())
191 scopedEnv->define(lambda.params[i], args[i]);
192 else if(i < args.size())
193 scopedEnv->define(lambda.defaultParams[i - lambda.params.size()].first,
194 args[i]);
195 else
196 scopedEnv->define(
197 lambda.defaultParams[i - lambda.params.size()].first,
198 evaluate(
199 lambda.defaultParams[i - lambda.params.size()].second.get(),
200 scopedEnv.get()));
201 }
202 try {
203 execute(lambda.body.get(), scopedEnv.get());
204 } catch(std::optional<std::any> &value) {
205 return value;
206 }
207 return std::make_optional<std::any>({});
208 };
209 return Callable{lambda.params.size(),
210 lambda.params.size() + lambda.defaultParams.size(),
211 lambdaFn,
212 std::make_shared<Environment>(env, true)};
213}

215std::optional<std::any>
216 Interpreter::visit(const Expression::Prototype &prototype,
217 Environment *env) {
218 std::shared_ptr<Environment> surroundingEnv{
219 std::make_shared<Environment>(env, true)};
220 std::shared_ptr<Environment> publicEnv{std::make_shared<Environment>()};
221 std::shared_ptr<Environment> privateEnv{std::make_shared<Environment>()};
222 if(prototype.parent) {
223 try {
224 std::any parent{env->get(prototype.parent.value())};
225 Prototypable parentPrototype{std::any_cast<Prototypable>(parent)};
226 privateEnv->copyOver(parentPrototype.privateEnv.get());
227 publicEnv->copyOver(parentPrototype.publicEnv.get());
228 surroundingEnv->define(Token{"parent", Token::Type::Identifier, true},
229 parentPrototype);
230 } catch(...) {
231 throw std::runtime_error{"Can only inherit from other prototypes."};
232 }
233 }
234 privateEnv->defineOrAssign(true);
235 publicEnv->defineOrAssign(true);
236 for(std::size_t i{0}; i < prototype.publicProperties.size(); i++)
237 execute(prototype.publicProperties[i].get(), publicEnv.get());
238 for(std::size_t i{0}; i < prototype.privateProperties.size(); i++)
239 execute(prototype.privateProperties[i].get(), privateEnv.get());
240 privateEnv->defineOrAssign(false);
241 publicEnv->defineOrAssign(false);
242 Callable constructor{0, 0, native::doNothing, surroundingEnv};
243 Prototypable anonymousPrototype{
244 constructor, surroundingEnv, publicEnv, privateEnv, nullptr};
245 anonymousPrototype.methodEnv = Environment::unionize(
246 {surroundingEnv.get(), publicEnv.get(), privateEnv.get()});
247 if(prototype.constructor)
248 anonymousPrototype.constructor = std::any_cast<Callable>(evaluate(
249 prototype.constructor.get(), anonymousPrototype.methodEnv.get()));
250 anonymousPrototype.methodEnv->define(
251 Token{"this", Token::Type::Identifier, true}, anonymousPrototype);
252 return anonymousPrototype;
253}

276std::optional<std::any> Interpreter::visit(const Expression::Get &get,
277 Environment *env) {
278 std::any object{evaluate(get.object.get(), env)};
279 std::any value;
280 try {
281 Prototypable prototype{std::any_cast<Prototypable>(object)};
282 try {
283 value = prototype.publicEnv->get(get.property);
284 try {
285 Callable callable{std::any_cast<Callable>(value)};
286 callable.fnEnv = prototype.methodEnv;
287 return callable;
288 } catch(std::bad_any_cast) {
289 return value;
290 }
291 } catch(std::runtime_error) {
292 try {
293 throw prototype.privateEnv->get(get.property);
294 } catch(std::runtime_error) {
295 throw std::runtime_error{"Property not found in prototype."};
296 } catch(std::any) {
297 throw std::runtime_error{"Requested property is private."};
298 }
299 }
300 } catch(std::bad_any_cast) {
301 throw std::runtime_error{"Can only receive properties from prototypes."};
302 }

371bool Interpreter::isTrue(const std::any &value) {
372 bool truth{false};
373 if(value.type() == typeid(bool) && std::any_cast<bool>(value))
374 truth = true;
375 else if(value.type() == typeid(long double) &&
376 std::any_cast<long double>(value) != 0.0)
377 truth = true;
378 else if(value.type() == typeid(std::string) &&
379 std::any_cast<std::string>(value) != "")
380 truth = true;
381 return truth;
382}

411std::any Interpreter::numericOperation(const long double left,
412 const Token::Type op,
413 const long double right) {
414 switch(op) {
415 case Token::Type::NotEqualTo: return left != right;
416 case Token::Type::EqualTo: return left == right;
417 case Token::Type::LessThan: return left < right;
418 case Token::Type::LessThanOrEqualTo: return left <= right;
419 case Token::Type::GreaterThan: return left > right;
420 case Token::Type::GreaterThanOrEqualTo: return left >= right;
421 case Token::Type::Asterisk: return left * right;
422 case Token::Type::Plus: return left + right;
423 case Token::Type::Dash: return left - right;
424 case Token::Type::ForwardSlash:
425 if(right == 0) throw std::runtime_error{"Attempted to divide by zero!"};
426 return left / right;
427 case Token::Type::Modulus:
428 if(right == 0)
429 throw std::runtime_error{
430 "Attempted to take remainder of division by zero!"};
431 return static_cast<long double>(fmod(left, right));
432 default: throw std::runtime_error("Not a supported binary operator.");
433 }
